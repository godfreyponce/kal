- [ ] **Step 1: Make `remove` report success and close the form from the caller**

```tsx
  async function remove(id: number): Promise<boolean> {
    if (deletingId !== null) return false;
    setDeletingId(id);
    setError(null);
    try {
      const res = await fetch(`/api/groceries/${id}`, { method: "DELETE" });
      if (!res.ok) {
        setError((await res.json().catch(() => ({}))).error ?? "Delete failed");
        return false;
      }
      startTransition(() => router.refresh());
      return true;
    } finally {
      setDeletingId(null);
    }
  }
```
